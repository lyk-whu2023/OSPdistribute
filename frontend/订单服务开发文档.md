# 前端订单服务开发文档

## 1. 概述

前端订单服务模块负责购物车管理、订单创建、订单列表、订单详情等相关功能。

### 1.1 技术栈

- **框架**: Vue 3 ^3.5.6
- **路由**: Vue Router ^4.3.0
- **UI 组件**: Element Plus ^2.8.6
- **HTTP 请求**: Axios ^1.7.7
- **状态管理**: Pinia ^2.1.7

### 1.2 文件结构

```
frontend/src/
├── api/
│   └── order.js              # 订单服务 API 接口
├── views/
│   ├── CartView.vue          # 购物车页面
│   ├── OrderSubmitView.vue   # 订单提交页面
│   ├── OrderListView.vue     # 订单列表页面
│   └── OrderDetailView.vue   # 订单详情页面
├── components/
│   └── order/                # 订单相关组件
│       ├── CartItem.vue       # 购物车商品项组件
│       └── OrderStatus.vue    # 订单状态组件
└── store/
    └── modules/
        └── cart.js           # 购物车状态管理
```

## 2. API 接口封装

### 2.1 创建 order.js

在 `frontend/src/api/order.js` 中创建订单服务 API 接口：

```javascript
import request from '@/utils/request'

/**
 * 获取购物车商品列表
 * @param {number} userId - 用户 ID
 * @returns {Promise}
 */
export function getCartItems(userId) {
  return request({
    url: `/api/cart/user/${userId}`,
    method: 'get'
  })
}

/**
 * 添加到购物车
 * @param {number} userId - 用户 ID
 * @param {number} productId - 商品 ID
 * @param {number} skuId - 规格 ID
 * @param {number} quantity - 数量
 * @param {number} price - 价格
 * @returns {Promise}
 */
export function addToCart(userId, productId, skuId, quantity, price) {
  return request({
    url: `/api/cart/user/${userId}/items`,
    method: 'post',
    params: { productId, skuId, quantity, price }
  })
}

/**
 * 更新购物车商品数量
 * @param {number} cartItemId - 购物车商品项 ID
 * @param {number} quantity - 数量
 * @returns {Promise}
 */
export function updateCartItem(cartItemId, quantity) {
  return request({
    url: `/api/cart/items/${cartItemId}`,
    method: 'put',
    params: { quantity }
  })
}

/**
 * 删除购物车商品
 * @param {number} cartItemId - 购物车商品项 ID
 * @returns {Promise}
 */
export function removeCartItem(cartItemId) {
  return request({
    url: `/api/cart/items/${cartItemId}`,
    method: 'delete'
  })
}

/**
 * 清空购物车
 * @param {number} userId - 用户 ID
 * @returns {Promise}
 */
export function clearCart(userId) {
  return request({
    url: `/api/cart/user/${userId}`,
    method: 'delete'
  })
}

/**
 * 获取用户订单列表
 * @param {number} userId - 用户 ID
 * @param {Object} params - 查询参数
 * @returns {Promise}
 */
export function getUserOrders(userId, params = {}) {
  return request({
    url: `/api/orders/user/${userId}`,
    method: 'get',
    params
  })
}

/**
 * 获取订单详情
 * @param {number} id - 订单 ID
 * @returns {Promise}
 */
export function getOrder(id) {
  return request({
    url: `/api/orders/${id}`,
    method: 'get'
  })
}

/**
 * 根据订单号获取订单
 * @param {string} orderNo - 订单号
 * @returns {Promise}
 */
export function getOrderByOrderNo(orderNo) {
  return request({
    url: `/api/orders/no/${orderNo}`,
    method: 'get'
  })
}

/**
 * 创建订单
 * @param {Object} data - 订单数据
 * @returns {Promise}
 */
export function createOrder(data) {
  return request({
    url: '/api/orders',
    method: 'post',
    params: {
      userId: data.userId,
      addressId: data.addressId,
      storeId: data.storeId
    },
    data: data.items
  })
}

/**
 * 更新订单状态
 * @param {number} id - 订单 ID
 * @param {string} status - 状态
 * @returns {Promise}
 */
export function updateOrderStatus(id, status) {
  return request({
    url: `/api/orders/${id}/status`,
    method: 'put',
    params: { status }
  })
}

/**
 * 取消订单
 * @param {number} id - 订单 ID
 * @returns {Promise}
 */
export function cancelOrder(id) {
  return request({
    url: `/api/orders/${id}/cancel`,
    method: 'put'
  })
}
```

## 3. 状态管理

### 3.1 创建购物车状态模块

在 `frontend/src/store/modules/cart.js` 创建购物车状态管理：

```javascript
import { defineStore } from 'pinia'
import { getCartItems, addToCart, updateCartItem, removeCartItem, clearCart } from '@/api/order'

export const useCartStore = defineStore('cart', {
  state: () => ({
    items: [],
    total: 0
  }),

  getters: {
    itemCount: (state) => state.items.length,
    totalPrice: (state) => {
      return state.items.reduce((sum, item) => {
        return sum + item.price * item.quantity
      }, 0)
    }
  },

  actions: {
    async loadCart(userId) {
      try {
        const response = await getCartItems(userId)
        this.items = response
        this.updateTotal()
      } catch (error) {
        console.error('加载购物车失败', error)
      }
    },

    async addToCart(userId, productId, skuId, quantity, price) {
      try {
        await addToCart(userId, productId, skuId, quantity, price)
        await this.loadCart(userId)
      } catch (error) {
        throw error
      }
    },

    async updateQuantity(cartItemId, quantity) {
      try {
        await updateCartItem(cartItemId, quantity)
        await this.loadCart()
      } catch (error) {
        throw error
      }
    },

    async removeItem(cartItemId) {
      try {
        await removeCartItem(cartItemId)
        await this.loadCart()
      } catch (error) {
        throw error
      }
    },

    async clear(userId) {
      try {
        await clearCart(userId)
        this.items = []
        this.total = 0
      } catch (error) {
        throw error
      }
    },

    updateTotal() {
      this.total = this.items.reduce((sum, item) => {
        return sum + item.price * item.quantity
      }, 0)
    }
  }
})
```

```vue
<script setup>
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { ElMessage, ElMessageBox } from 'element-plus'
import { useUserStore } from '@/store/modules/user'
import { useCartStore } from '@/store/modules/cart'
import { getCartItems, removeCartItem, clearCart } from '@/api/order'

const router = useRouter()
const userStore = useUserStore()
const cartStore = useCartStore()

const cartItems = ref([])

const totalPrice = computed(() => {
  return cartItems.value.reduce((sum, item) => {
    return sum + item.price * item.quantity
  }, 0)

  await loadCart()
})

  try {
    const response = await getCartItems(userStore.userId)
    cartItems.value = response

    ElMessage.error('加载购物车失败')
  }
}

  try {
    await updateCartItem(row.id, row.quantity)
    ElMessage.success('更新成功')
  } catch (error) {
    ElMessage.error('更新失败')
  }
}

const handleRemove = async (row) => {
  try {
    await ElMessageBox.confirm('确定删除该商品吗？', '提示', {
      type: 'warning'
    })
    await removeCartItem(row.id)
    ElMessage.success('删除成功')
    await loadCart()
  } catch (error) {
    if (error !== 'cancel') {
      ElMessage.error('删除失败')
    }
  }
}

const handleClearCart = async () => {
  try {
    await ElMessageBox.confirm('确定清空购物车吗？', '提示', {
      type: 'warning'
    })
    await clearCart(userStore.userId)
    ElMessage.success('清空成功')
    await loadCart()
  } catch (error) {
    if (error !== 'cancel') {
      ElMessage.error('清空失败')
    }
  }
}

const handleCheckout = () => {
  if (cartItems.value.length === 0) {
    ElMessage.warning('购物车为空')
    return
  }
  
  router.push({
    path: '/order/submit',
    query: {
      items: JSON.stringify(cartItems.value.map(item => ({
        productId: item.productId,
        skuId: item.skuId,
        quantity: item.quantity,
        price: item.price
      })))
    }
  })
}

const goToProducts = () => {
  router.push('/products')
}
</script>
```

```vue
import { ref, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { ElMessage } from 'element-plus'

import { getUserAddresses } from '@/api/user'

const addressList = ref([])
const orderItems = ref([])

}

const loadOrderItems = () => {
  const itemsParam = route.query.items
  if (itemsParam) {
    orderItems.value = JSON.parse(itemsParam)
  }
}

const handleSubmitOrder = async () => {
  if (!selectedAddressId.value) {
    ElMessage.warning('请选择收货地址')
    return
  }

  if (orderItems.value.length === 0) {
    ElMessage.warning('商品清单为空')
    return
  }

  try {
    submitting.value = true
    
    const orderData = {
      userId: userStore.userId,
      addressId: selectedAddressId.value,
      storeId: orderItems.value[0]?.storeId || 1,
      items: orderItems.value.map(item => ({
        productId: item.productId,
        skuId: item.skuId,
        productName: item.productName,
        skuName: item.skuName,
        price: item.price,
        quantity: item.quantity
      }))
    }

    const order = await createOrder(orderData)
    
    ElMessage.success('订单创建成功')
    router.push(`/order/${order.id}`)
  } catch (error) {
    ElMessage.error(error.response?.data?.message || '订单创建失败')
  } finally {
    submitting.value = false
  }
}
</script>
```

## 5. 路由配置

在 `frontend/src/router/index.js` 中添加订单相关路由：

```javascript
import { createRouter, createWebHistory } from 'vue-router'
import { useUserStore } from '@/store/modules/user'

const routes = [
  {
    path: '/cart',
    name: 'Cart',
    component: () => import('@/views/CartView.vue'),
    meta: { requiresAuth: true }
  },
  {
    path: '/order/submit',
    name: 'OrderSubmit',
    component: () => import('@/views/OrderSubmitView.vue'),
    meta: { requiresAuth: true }
  },
  {
    path: '/orders',
    name: 'OrderList',
    component: () => import('@/views/OrderListView.vue'),
    meta: { requiresAuth: true }
  },
  {
    path: '/order/:id',
    name: 'OrderDetail',
    component: () => import('@/views/OrderDetailView.vue'),
    meta: { requiresAuth: true }
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

router.beforeEach((to, from, next) => {
  const userStore = useUserStore()
  
  if (to.meta.requiresAuth && !userStore.isLoggedIn) {
    next({ name: 'Auth', query: { redirect: to.fullPath } })
  } else {
    next()
  }
})

export default router
```
