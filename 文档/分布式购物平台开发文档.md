# 分布式在线购物平台开发文档

## 1. 项目概述

本项目是一个基于分布式架构的在线购物平台，提供商品浏览、购物车管理、订单处理、用户管理等核心功能，同时为管理员提供完善的后台管理系统。

```java
// 短信服务
@Service
public class SmsService {
    @Autowired
    private AliyunSmsClient aliyunSmsClient;
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    public void sendVerificationCode(String phone) {
        // 生成6位验证码
        String code = String.valueOf(new Random().nextInt(900000) + 100000);
        // 发送短信
        aliyunSmsClient.sendSms(phone, code);
        // 缓存验证码，有效期5分钟
        redisTemplate.opsForValue().set("sms:code:" + phone, code, 5, TimeUnit.MINUTES);
    }
    
    public boolean verifyCode(String phone, String code) {
        String cachedCode = redisTemplate.opsForValue().get("sms:code:" + phone);
        return code != null && code.equals(cachedCode);
    }
}

// 登录控制器
@RestController
@RequestMapping("/api/auth")
public class AuthController {
    @Autowired
    private UserService userService;
    @Autowired
    private SmsService smsService;
    @Autowired
    private JwtUtil jwtUtil;
    
    @PostMapping("/sms/login")
    public Map<String, Object> smsLogin(@RequestBody Map<String, String> body) {
        String phone = body.get("phone");
        String code = body.get("code");
        
        // 验证验证码
        if (!smsService.verifyCode(phone, code)) {
            throw new IllegalArgumentException("验证码错误");
        }
        
        // 查找或创建用户
        UserDTO user = userService.findByPhone(phone);
        if (user == null) {
            user = userService.createUserByPhone(phone);
        }
        
        // 生成token
        String token = jwtUtil.createToken(user.getId(), user.getUsername(), user.getRole());
        
        Map<String, Object> resp = new HashMap<>();
        resp.put("token", token);
        resp.put("user", user);
        return resp;
    }
}
```

```java
// 秒杀服务
@Service
public class SeckillService {
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    @Autowired
    private SeckillProductMapper seckillProductMapper;
    @Autowired
    private KafkaTemplate<String, Object> kafkaTemplate;
    
    // 初始化秒杀商品库存到 Redis
    @PostConstruct
    public void initSeckillStock() {
        List<SeckillProduct> products = seckillProductMapper.selectList(null);
        for (SeckillProduct product : products) {
            redisTemplate.opsForValue().set("seckill:stock:" + product.getId(), product.getStock());
        }
    }
    
    // 秒杀抢购
    public boolean seckill(Long productId, Long userId) {
        // 检查是否已经抢购过
        if (redisTemplate.hasKey("seckill:order:" + userId + ":" + productId)) {
            return false;
        }
        
        // 减库存
        Long stock = redisTemplate.opsForValue().decrement("seckill:stock:" + productId);
        if (stock < 0) {
            // 库存不足，回滚
            redisTemplate.opsForValue().increment("seckill:stock:" + productId);
            return false;
        }
        
        // 标记用户已抢购
        redisTemplate.opsForValue().set("seckill:order:" + userId + ":" + productId, "1", 24, TimeUnit.HOURS);
        
        // 发送消息到 Kafka，异步处理订单
        SeckillMessage message = new SeckillMessage();
        message.setProductId(productId);
        message.setUserId(userId);
        kafkaTemplate.send("seckill-topic", message);
        
        return true;
    }
```

### 7.9 店铺关注实现

```java
// 店铺服务
@Service
public class StoreService {
    @Autowired
    private StoreMapper storeMapper;
    @Autowired
    private StoreFollowMapper storeFollowMapper;
    
    public void followStore(Long storeId, Long userId) {
        // 检查是否已经关注
        StoreFollow follow = storeFollowMapper.selectOne(
            new QueryWrapper<StoreFollow>()
                .eq("store_id", storeId)
                .eq("user_id", userId)
        );
        
        if (follow == null) {
            // 添加关注
            follow = new StoreFollow();
            follow.setStoreId(storeId);
            follow.setUserId(userId);
            follow.setFollowTime(new Date());
            storeFollowMapper.insert(follow);
            
            // 更新店铺关注数
            storeMapper.update(
                new UpdateWrapper<Store>()
                    .setSql("follow_count = follow_count + 1")
                    .eq("id", storeId)
            );
        }
    }
    
    public void unfollowStore(Long storeId, Long userId) {
        // 取消关注
        storeFollowMapper.delete(
            new QueryWrapper<StoreFollow>()
                .eq("store_id", storeId)
                .eq("user_id", userId)
        );
        
        // 更新店铺关注数
        storeMapper.update(
            new UpdateWrapper<Store>()
                .setSql("follow_count = follow_count - 1")
                .eq("id", storeId)
        );
    }
    
    public List<StoreDTO> getFollowedStores(Long userId) {
        // 查询用户关注的店铺
        List<StoreFollow> follows = storeFollowMapper.selectList(
            new QueryWrapper<StoreFollow>()
                .eq("user_id", userId)
        );
        
        List<Long> storeIds = follows.stream()
            .map(StoreFollow::getStoreId)
            .collect(Collectors.toList());
        
        if (storeIds.isEmpty()) {
            return Collections.emptyList();
        }
        
        // 查询店铺信息
        List<Store> stores = storeMapper.selectBatchIds(storeIds);
        return stores.stream().map(store -> {
            StoreDTO dto = new StoreDTO();
            BeanUtils.copyProperties(store, dto);
            return dto;
        }).collect(Collectors.toList());
    }
}
```

### 7.10 统计功能实现

```java
// 统计服务
@Service
public class StatsService {
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private UserMapper userMapper;
    @Autowired
    private ProductMapper productMapper;
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    // 销售统计
    public SalesStatsDTO getSalesStats(Date startDate, Date endDate) {
        // 从数据库查询
        List<Order> orders = orderMapper.selectList(
            new QueryWrapper<Order>()
                .between("create_time", startDate, endDate)
                .eq("status", "completed")
        );
        
        // 计算销售额
        BigDecimal totalSales = orders.stream()
            .map(Order::getAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        // 计算订单数
        int orderCount = orders.size();
        
        SalesStatsDTO stats = new SalesStatsDTO();

        stats.setStartDate(startDate);

    }

    public RealTimeStatsDTO getRealTimeStats() {

            ? (Long) redisTemplate.opsForValue().get("stats:seckill:count") 
            : 0;
        
        Long blogViewCount = redisTemplate.opsForValue().get("stats:blog:view:count") != null

            : 0;
        
        RealTimeStatsDTO stats = new RealTimeStatsDTO();
        stats.setSeckillCount(seckillCount);
```
